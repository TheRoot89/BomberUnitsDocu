66  public void ExplodeGrid() {
67  // Check GameObj map for affected Environment Objects
68  objAffected.Clear();
69  gridCellsAffected.Clear();
70  Vector3 posUnderTest;
71  // +X
72  for (int d = 1; d <= range; d++) {
73  posUnderTest = new Vector3(Mathf.FloorToInt(transform.position.x) + d, 0, Mathf.FloorToInt(transform.position.z));
74  if (!TestPosition(posUnderTest))
75  break;
76  }
77  //-X
78  for (int d = 1; d <= range; d++) {
79  posUnderTest = new Vector3(Mathf.FloorToInt(transform.position.x) - d, 0, Mathf.FloorToInt(transform.position.z));
80  if (!TestPosition(posUnderTest))
81  break;
82  }
83  //+Z
84  for (int d = 1; d <= range; d++) {
85  posUnderTest = new Vector3(Mathf.FloorToInt(transform.position.x), 0, Mathf.FloorToInt(transform.position.z) + d);
86  if (!TestPosition(posUnderTest))
87  break;
88  }
89  //-Z
90  for (int d = 1; d <= range; d++) {
91  posUnderTest = new Vector3(Mathf.FloorToInt(transform.position.x), 0, Mathf.FloorToInt(transform.position.z) - d);
92  if (!TestPosition(posUnderTest))
93  break;
94  }
95 
96  // Get players and test whether any of them is in range:
97  foreach (Player p in PlayerManager.Get().GetPlayers()) {
98  Vector3 playerGridPos = new Vector3(Mathf.Floor(p.playerFigure.transform.position.x), 0, Mathf.Floor(p.playerFigure.transform.position.z));
99  if (gridCellsAffected.Contains(playerGridPos))
100  objAffected.Add(p.playerFigure);
101  }
102 
103  foreach (Vector3 v in gridCellsAffected)
104  Instantiate(subExplosionEffect, v + new Vector3(0.5f, 1, 0.5f), Quaternion.identity);
105 
106  foreach (GameObject c in objAffected)
107  Hit(c);
108 
109  if (range >= bigExplMinRange)
110  Instantiate(coreExplosionEffectBIG, transform.position + Vector3.up, Quaternion.identity);
111  else
112  Instantiate(coreExplosionEffect, transform.position + Vector3.up, Quaternion.identity);
113

154  private void BreakAndThrow(GameObject obj) {
155  // Break:
156  Breakable breakable = obj.GetComponent<Breakable>();
157  if (breakable == null)
158  return;
159 
160  parts = breakable.Break();
161 
162  // Throw:
163  foreach (GameObject part in parts) {
164  if (part.GetComponent<Rigidbody>() == null) {
165  part.AddComponent<Rigidbody>();
166  part.GetComponent<Rigidbody>().mass = splinterBaseWeight;
167  }
168  // The perceived explosion force is also dependent on the expl range. Hence, with increasing range, the force has to be adapted.
169  /// \todo: Add rotation to exploded parts. Either here with rng or maybe via setting the center of
170  /// mass off a bit for these objects during modelling?
171  part.GetComponent<Rigidbody>().AddExplosionForce(baseForce / (0.8f * range) + range, transform.position, range, upFactor, ForceMode.Impulse);
172  if (!partsCollision && part.GetComponent<BoxCollider>() != null)
