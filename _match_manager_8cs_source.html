146  /// Each players current status is held within its player information.
147  /// A reference to the winnings player is filled if the match result was not a draw.
148  /// The player deaths are synchronized over clients, so the evaluation needs not to.
149  private void CheckForMatchEnd() {
150  if (state != MatchState.running) {
151  return;
152  }
153 
154  int playersLeft = 0;
155  winningPlayer = null;
156  List<Player> players = PlayerManager.Get().GetPlayers();
157  foreach (Player p in players) {
158  if (p.state == Player.State.playing) {
159  winningPlayer = p;
160  playersLeft++;
161  }
162  }
163  // After the iteration 'winningPlayer' is set to the last survivor, if any.
164 
165  if (playersLeft <= 1) {
166  state = MatchState.gameOver;
167  GameManager.GetSM().GameOver();
168  }
169  }
