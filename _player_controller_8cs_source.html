134  private void SpawnBombIfFree()
135  {
136  // Grid snapping: (aka 1-unit-grid, not for y which is determined by the playerFigures' bombSpawn object)
137  Vector3 gridSnappedSpawn = new Vector3(Mathf.Floor(bombSpawnPos.position.x) + 0.5f,
138  bombSpawnPos.position.y,
139  Mathf.Floor(bombSpawnPos.position.z) + 0.5f);
140  // Check if a bomb is already placed here:
141  Collider[] collidingObjects = Physics.OverlapBox(gridSnappedSpawn, new Vector3(0.5f, 2, 0.5f));
142  foreach (Collider c in collidingObjects) //for each found object
143  {
144  if (c.name.Contains("Bomb"))
145  return;
146  }
147  // Spawn: (also on the clients)
148  GameObject bomb = Instantiate(bombPrefab, gridSnappedSpawn, bombSpawnPos.rotation);
149  bomb.GetComponent<Explodeable>().range = bombRange;
150  bomb.GetComponent<Explodeable>().Light(bombCountdown);
151  }
