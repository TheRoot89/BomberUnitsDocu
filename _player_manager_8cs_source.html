247  /// Add a new player given by the UI, given from loading or handed by connecting clients in the same
248  /// match.
249  private bool AddPlayer(Player p) {
250  if (numPlayers >= GameSettings.Get().maxTotPlayers) {
251  return false;
252  }
253 
254  if (p.IsLocal() && (numLocPlayers >= GameSettings.Get().maxLocPlayers)) {
255  return false;
256  }
257 
258  // Else the player can be added:
259  players.Add(p);
260 
261  // Count the addition and mark used local resources:
262  if (p.IsLocal()) {
263  numLocPlayers++;
264  p.input.taken = true;
265  p.color.taken = true;
266  }
267  numPlayers++;
268 
269  if (lobbyPanelController != null) {
270  lobbyPanelController.UpdateAddPlayerBtnAvailability();
271  }
272  return true;
273  }
274 
275 
276  /// Player removal can be caused by:
277  /// * a user action: remove the player for which deletion was called from the UI. This can happen when
278  /// this or a remote client removes a player and is transmitted via an event.
279  /// * or disconnect of this client: RemoveAllRemotePlayers of any clients which were present before.
280  /// * or disconnect another client: RemoveAllRemotePlayersOf the particular client.
281  private void RemovePlayer(Player p) {
282  // Also remove slot:
283  if (p.playerSlot != null) {
284  Destroy(p.playerSlot.gameObject);
285  }
286 
287  if (p.type == Player.Type.local) {
288  numLocPlayers--;
289  p.input.taken = false;
290  p.color.taken = false;
291  }
292  numPlayers--;
293 
294  players.Remove(p);
295 
296  if (lobbyPanelController != null) {
297  lobbyPanelController.UpdateAddPlayerBtnAvailability();
298  }
299 
300  // Safety check:
301  if (numPlayers != players.Count) {
302  Debug.LogError("Mismatch in player number tracking!");
303  }
304  }
