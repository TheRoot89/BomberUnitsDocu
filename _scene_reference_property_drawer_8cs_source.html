231  /// Check if the build settings asset is readonly.
232  /// Caches value and only queries state a max of every 'minCheckWait' seconds.
233  public static bool IsReadOnly() {
234  var curTime = Time.realtimeSinceStartup;
235  var timeSinceLastCheck = curTime - lastTimeChecked;
236 
237  if (!(timeSinceLastCheck > minCheckWait)) return cachedReadonlyVal;
238 
239  lastTimeChecked = curTime;
240  cachedReadonlyVal = QueryBuildSettingsStatus();
241 
242  return cachedReadonlyVal;
243  }
244 
245  /// A blocking call to the Version Control system to see if the build settings asset is readonly.
246  /// Use BuildSettingsIsReadOnly for version that caches the value for better responsiveness.
247  private static bool QueryBuildSettingsStatus() {
248  // If no version control provider, assume not readonly
249  if (!Provider.enabled) return false;
250 
251  // If we cannot checkout, then assume we are not readonly
252  if (!Provider.hasCheckoutSupport) return false;
253 
254  //// If offline (and are using a version control provider that requires checkout) we cannot edit.
255  //if (UnityEditor.VersionControl.Provider.onlineState == UnityEditor.VersionControl.OnlineState.Offline)
256  // return true;
257 
258  // Try to get status for file
259  var status = Provider.Status("ProjectSettings/EditorBuildSettings.asset", false);
260  status.Wait();
261 
262  // If no status listed we can edit
263  if (status.assetList == null || status.assetList.Count != 1) return true;
264 
265  // If is checked out, we can edit
266  return !status.assetList[0].IsState(Asset.States.CheckedOutLocal);
267  }

269  /// For a given Scene Asset object reference, extract its build settings data, including buildIndex.
270  public static BuildScene GetBuildScene(Object sceneObject) {
271  var entry = new BuildScene {
272  buildIndex = -1,
273  assetGUID = new GUID(string.Empty)
274  };
275 
276  if (sceneObject as SceneAsset == null) return entry;
277 
278  entry.assetPath = AssetDatabase.GetAssetPath(sceneObject);
279  entry.assetGUID = new GUID(AssetDatabase.AssetPathToGUID(entry.assetPath));
280 
281  for (var index = 0; index < EditorBuildSettings.scenes.Length; ++index) {
282  if (!entry.assetGUID.Equals(EditorBuildSettings.scenes[index].guid)) continue;
283 
284  entry.scene = EditorBuildSettings.scenes[index];
285  entry.buildIndex = index;
286  return entry;
287  }
288 
289  return entry;
290  }
