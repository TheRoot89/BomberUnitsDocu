135  /// Moves the given element into view from the given direction, but only if not already shown.

137  public void Show(ScreenDir fromDir)

146  /// Gets the "visible" boolean parameter from the Animator
147  private bool isShown() {
148  if (anim) {
149  return anim.GetBool(visibleBoolParamName);
150  } else {
151  Debug.LogWarningFormat( "UIMenuElement of {0}: was not initialized before call to Animator.",
152  this.gameObject.name);
153  return false;
154  }

158  /// Coroutine allowing to delay the show animation as specified in the settings.
159  /// The element and optional parent container are activated and moved to the front.
160  IEnumerator ShowElementCo(ScreenDir fromDir)
161  {
162  yield return new WaitForSeconds(showDelay);
163 
164  // Activate and move to the Front: Container
165  if (container != null) {
166  container.SetActive(true);
167  container.transform.SetAsLastSibling();
168  }
169 
170  // Activate and move to the Front: this
171  gameObject.SetActive(true);
172  transform.SetAsLastSibling();
173 
174  // Start the open animation:
175  anim.SetInteger(directionIntParamName, (int)fromDir);
176  anim.SetBool(visibleBoolParamName, true);
177 
178  curShownElements.Add(this); // Track this elem. as currently shown
179  }
180 
181 
182  /// Coroutine waiting until the Closing animation is finished to disable the object hierarchy.
183  IEnumerator DisableAfterAnimFinishedCo(ScreenDir toDir)
184  {
185  bool closedStateReached = false;
186  bool wantToClose = true;
187  while (!closedStateReached && wantToClose)
188  {
189  if (!anim.IsInTransition(0))
190  closedStateReached = anim.GetCurrentAnimatorStateInfo(0).IsName(screenDirHiddenStateNames[toDir]);
191 
192  wantToClose = !isShown();
193 
194  yield return new WaitForEndOfFrame();
195  }
196 
197  if (wantToClose)
198  {
199  if (container != null) {
200  container.SetActive(false);
201  }
202  gameObject.SetActive(false);
203  }
204  }
