63  /// Hides everything not also on the new page and shows that page instead, elements on both pages are
64  /// not animated.
65  public void ReplaceBy(UIMenuPage other)
66  {
67  if (!isShown)
68  {
69  Debug.LogWarningFormat("UIMenuPage {0}: Trying to remove although not shown. Check your logic.",
70  this.name);
71  return;
72  }
73 
74  foreach (UIMenuElement ownElem in pageElements)
75  {
76  if (!other.Contains(ownElem))
77  ownElem.Hide();
78  }
79  this.isShown = false;
80  lastHiddenPage = this;
81 
82  other.Show();
83  }

102  /// Closes everything else before showing this page, elements already shown are not animated.
103  /// Idea: return a backup:
104  /// \code
105  /// UIMenuPage backup = new UIMenuPage(curShownElements.ToArray());
106  /// ...
107  /// return backup;
108  /// \endcode
109  public void ShowExclusively()
110  {
111  foreach (UIMenuPage p in GetAllShownPages())
112  {
113  if (p == this)
114  { // Do not try and close this page, as it should be shown.
115  continue;
116  }
117 
118  // Close all other pages' elements not needed:
119  foreach (UIMenuElement e in p.pageElements)
120  {
121  if (!this.Contains(e))
122  {
123  e.Hide();
124  }
125  }
126  p.isShown = false;
127  }
128 
129  Show();
130  }

249  public UIMenuElement[] ElemNotAlsoInOther(UIMenuPage other)
250  {
251  List<UIMenuElement> ownElemNotInOther = new List<UIMenuElement>();
252  foreach (UIMenuElement ownElem in pageElements)
253  {
254  bool found = false;
255  foreach (UIMenuElement otherElem in other.pageElements)
256  {
257  if (ownElem == otherElem)
258  {
259  found = true;
260  break;
261  }
262  }
263  if (!found) ownElemNotInOther.Add(ownElem);
264  }
265 
266  return ownElemNotInOther.ToArray();
