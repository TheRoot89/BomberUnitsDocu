279  private void ShowThrowableTrajectoryUI() {

330  private void PlaceHeldBomb() {
331  Bomb b = heldBomb.GetComponent<Bomb>();
332 
333  Tile tile = Tile.Underneath(heldBomb.transform.position);
334  if (tile != null && tile.GetTileObject() == null) { // tile is there and free
335  tile.SetTileObject(heldBomb.gameObject); // For reference that there is now an object on the tile.
336  b.CenterOn(tile.transform.position);
337  }
338  else {
339  b.EnableFreePhysics();
340  }
341 
342  b.LightFuze(); // This will start a synced explosion timer on all clients using server time.
343  heldBomb = null; // we are no longer holding the bomb, remove.
344  }
