73  /// The Break function returns a list of GOs which represent the broken parts of the object this script
74  /// is attached to.
75  /// There are two ways to use this:
76  /// a) Set a different 'broken' version of the model in the 'brokenObjectPrefab' slot. The script
77  /// expects to find the broken pieces as first-level children directly underneath the prefab object.
78  /// b) Add children of this GO to the 'brokenParts' list.
79  /// All parts are expected to have suitably weighted rigidbodies attached, which may be disabled though.
80  ///
81  /// If you do both, option a) will be preferred by the script.
82  ///
83  /// Also breaking an object may spawn loot.
84  private List<GameObject> Break() {
85  List<GameObject> partsReturned = new List<GameObject>();
86 
87  // If this GO is not active then don't do anything:
88  if (!gameObject.activeInHierarchy) {
89  return partsReturned;
90  }
91 
92  // Determine whether option a) or b) is used. See descr. above.
93  // And get the list of GOs accordingly.
94  bool replacedByBrokenModel = brokenObjectPrefab != null;
95  if (replacedByBrokenModel) {
96  GameObject brokenVersion = Instantiate(brokenObjectPrefab, transform.position, transform.rotation);
97  for (int childIndex = 0; childIndex < brokenVersion.transform.childCount; childIndex++) {
98  partsReturned.Add(brokenVersion.transform.GetChild(childIndex).gameObject);
99  }
100  this.gameObject.SetActive(false);
101  }
102  else {
103  partsReturned.AddRange(brokenParts);
104  }
105 
106  // Spawn loot if applicable: /// \todo maybe move this responsibility. it is very hidden here.
107  LootSpawner lootSpawner = GetComponent<LootSpawner>();
108  if (lootSpawner != null) {
109  lootSpawner.RollDropLoot();
110  }
111 
112  return partsReturned;
113  }

116  /// Activates Physics for the piece and applies the damage force.
117  private void Throw(GameObject piece, Explodeable.Damage dmg) {
118  Rigidbody rb = piece.GetComponent<Rigidbody>();
119  if (rb == null) {
120  Debug.LogWarning("BreakAndThrow: The piece " + piece.name + " has no rigidbody!");
121  return;
122  }
123  else {
124  rb.isKinematic = false; // Kinematic = true does not take part in physics, so deactivate
125  }
126 
127  // Prepare collider so collisions are possible for this piece (or switched off if set above):
128  Collider col = piece.GetComponentInChildren<Collider>();
129  if (piecesShallCollide) {
130  if (col == null) {
131  col = piece.AddComponent<BoxCollider>();
132  }
133  col.enabled = true;
134  }
135  else {
136  if (col != null) {
137  col.enabled = false;
138  }
139  }
140 
141  // We want to increase the force acting on objects remote to the origin, so we half the distance:
142  Vector3 forceOrigin = (transform.position + dmg.origin) / 2;
143  piece.GetComponent<Rigidbody>().AddExplosionForce(dmg.force, forceOrigin, dmg.forceRange,
144  dmg.upFactor, ForceMode.Impulse);
145  }
