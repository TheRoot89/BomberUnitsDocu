58  /// The standard explosion of a bomb has the shape of a hor and vert cross with a given range.
59  /// For this explosion type, there has to be a tile underneath!
60  public void ExplodeGrid(Tile origin, int gridRange) {
61  List<Tile> affectedTiles = new List<Tile>();
62  AoEScheme aoe = new CrossAoEScheme(origin, gridRange);
63  aoe.Evaluate(ref affectedTiles);
64  foreach (Tile t in affectedTiles) {
65  if (t == origin) {
66  InstantiateCoreEffect(gridRange); // implicit conversion from int to float
67  }
68  else {
69  Instantiate(subExplosionEffect, t.transform.position + Vector3.up * 0.75f, Quaternion.identity);
70  }
71  }
72 
73  // Find tile objects affected:
74  foreach (Tile t in affectedTiles) {
75  GameObject tileObj = t.GetTileObject();
76  if (tileObj != null) {
77  Destructible d = tileObj.GetComponent<Destructible>();
78  if (d != null) {
79  d.Hit(damage);
80  }
81  }
82  }
83 
84  // Find player figures affected:
85  // Players are treated as point-like at this point and are only ever present on a single tile.
86  foreach (Player p in Player.GetAll().Values) {
87  if (p.GetState() == Player.State.playing) { // only playing players have figures of course
88  Tile playerTile = Tile.Underneath(p.playerFigure.transform.position + Vector3.up);
89  if (affectedTiles.Contains(playerTile)) {
90  p.playerFigure.GetComponent<Destructible>().Hit(damage);
91  }
92  }
93  }
94  }
