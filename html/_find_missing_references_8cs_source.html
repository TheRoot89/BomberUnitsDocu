40  [MenuItem(MENU_ROOT + "Search in assets", false, 52)]
41  public static void FindMissingReferencesInAssets() {
42  var allAssets = AssetDatabase.GetAllAssetPaths().Where(path => path.StartsWith("Assets/")).ToArray();
43  var objs = allAssets.Select(a => AssetDatabase.LoadAssetAtPath(a, typeof(GameObject)) as GameObject).Where(a => a != null).ToArray();

48  private static void FindMissingReferences(string context, GameObject[] gameObjects) {
49  if (gameObjects == null) {
50  return;
51  }
52 
53  bool nothingFound = true;
54  foreach (var go in gameObjects) {
55  var components = go.GetComponents<Component>();
56 
57  foreach (var component in components) {
58  // Missing components will be null, we can't find their type, etc.
59  if (!component) {
60  Debug.LogErrorFormat(go, $"Missing Component {0} in GameObject: {1}", component.GetType().FullName, GetFullPath(go));
61 
62  continue;
63  }
64 
65  SerializedObject so = new SerializedObject(component);
66  var sp = so.GetIterator();
67 
68  var objRefValueMethod = typeof(SerializedProperty).GetProperty("objectReferenceStringValue",
69  BindingFlags.Instance | BindingFlags.NonPublic | BindingFlags.Public);
70 
71  // Iterate over the components' properties.
72  while (sp.NextVisible(true)) {
73  if (sp.propertyType == SerializedPropertyType.ObjectReference) {
74  string objectReferenceStringValue = string.Empty;
75 
76  if (objRefValueMethod != null) {
77  objectReferenceStringValue = (string)objRefValueMethod.GetGetMethod(true).Invoke(sp, new object[] { });
78  }
79 
80  if (sp.objectReferenceValue == null
81  && (sp.objectReferenceInstanceIDValue != 0 || objectReferenceStringValue.StartsWith("Missing"))) {
82  nothingFound = false;
83  ShowError(context, go, component.GetType().Name, ObjectNames.NicifyVariableName(sp.name));
84  }
85  }
86  }
87  }
88  }
89 
90  if (nothingFound) {
91  Debug.Log("No missing references were found! :)");
92  }
93  }
94 
95  private static GameObject[] GetSceneObjects() {
96  // Use this method since GameObject.FindObjectsOfType will not return disabled objects.
97  return Resources.FindObjectsOfTypeAll<GameObject>()
98  .Where(go => string.IsNullOrEmpty(AssetDatabase.GetAssetPath(go))
99  && go.hideFlags == HideFlags.None).ToArray();
100  }

102  private static void ShowError(string context, GameObject go, string componentName, string propertyName) {
103  var ERROR_TEMPLATE = "Missing Ref in: [{3}]{0}. Component: {1}, Property: {2}";
104 
105  Debug.LogError(string.Format(ERROR_TEMPLATE, GetFullPath(go), componentName, propertyName, context), go);
106  }
107 
108  private static string GetFullPath(GameObject go) {
109  return go.transform.parent == null
110  ? go.name
111  : GetFullPath(go.transform.parent.gameObject) + "/" + go.name;
112  }
