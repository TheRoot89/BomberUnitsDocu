23  /// Provides Fisher-Yates in-place shuffling for any list-interface object.
24  public static void Shuffle<T>(this IList<T> list) {
25  int n = list.Count;
26  while (n > 1) {
27  n--;
28  int k = Mathf.FloorToInt(Random.value * n);
29  T value = list[k];
30  list[k] = list[n];
31  list[n] = value;
32  }
33  }
