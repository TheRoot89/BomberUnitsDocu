31  /// Provides Fisher-Yates in-place shuffling for any list-interface object.
32  public static void Shuffle<T>(this IList<T> list) {
33  int n = list.Count;
34  while (n > 1) {
35  n--;
36  int k = Mathf.FloorToInt(UnityEngine.Random.value * n);
37  T value = list[k];
38  list[k] = list[n];
39  list[n] = value;
40  }
41  }

139  float totalWeight = sequence.Sum(weightSelector);
140  float rndSelectedWeight = UnityEngine.Random.value * totalWeight;
141 
142  float cumWeightOfCheckedItems = 0;
143  foreach (var item in from weightedItem in sequence select new { Value = weightedItem, Weight = weightSelector(weightedItem) }) {
144  cumWeightOfCheckedItems += item.Weight;
145 
146  // If we've hit or passed the weight we are after for this item then it's the one we want:
147  if (cumWeightOfCheckedItems >= rndSelectedWeight) {
148  return item.Value;
