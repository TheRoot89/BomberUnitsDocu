144  /// Calculates the size of thumbnails and padding
145  private void ResizeThumbnails() {
146  float percentageUsedByCells = 1 - (vertSpacingPercent / 100) * (fixedNumRows + 1);
147 
148  float thumbWidth = gridLayoutChild.cellSize.x;
149  float thumbHeight = gridLayoutChild.cellSize.y;
150 
151  float totalHeightOriginallyNeeded = thumbHeight * fixedNumRows / percentageUsedByCells;
152  float totalHeightAvail = GetComponent<RectTransform>().rect.height;
153  float scaleFactor = totalHeightAvail / totalHeightOriginallyNeeded;
154  int vertSpacing = Mathf.RoundToInt(totalHeightAvail * vertSpacingPercent / 100);
155 
156  gridLayoutChild.cellSize = new Vector2(thumbWidth, thumbHeight) * scaleFactor;
157  gridLayoutChild.spacing = new Vector2(vertSpacing, vertSpacing);
158  gridLayoutChild.padding = new RectOffset(vertSpacing, vertSpacing, vertSpacing, vertSpacing);
159  }

166  int colCount = Mathf.FloorToInt(levelThumbnails.Count / (float)fixedNumRows);
