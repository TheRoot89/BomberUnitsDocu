28  /// A vertex is referenced by other mesh code via its position in the vertex array.
29  /// In writing all Vertices into a common list, we need to update the indices triangles refer to.
30  /// We thus keep a counter which offsets a new set of incoming vertices with the nr of already present
31  /// entries.
32  private void ExtractVerticesAndTris(List<GameObject> objectsWithMeshColliders) {
33  int vertexCounter = 0;
34  foreach (GameObject obj in objectsWithMeshColliders) {
35  foreach (MeshCollider mc in obj.GetComponentsInChildren<MeshCollider>()) {
36  // Add triangles: (Added first because the vertex counter is still zero here.)
37  foreach (int triangleCorner in mc.sharedMesh.triangles) {
38  allTris.Add(triangleCorner + vertexCounter);
39  }
40 
41  // Add Vertices (this updates the triangle counter)
42  foreach (Vector3 vertex in mc.sharedMesh.vertices) {
43  Vector3 vertInGlobCoord = mc.transform.TransformVector(vertex) + mc.transform.position;
44  vertInGlobCoord = obj.transform.TransformVector(vertInGlobCoord);
45  allVertices.Add(vertInGlobCoord);
46  vertexCounter++;
47  }
48  }
49  }
