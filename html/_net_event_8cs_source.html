21  /// This is a convenience wrapper for Photons networked event system.
22  /// It primarily allows for quick raising and registering to these events, removing some low-level actions.
23  /// They can be set up for any Type of data package implementing INetDataPkg for serialization.
24  /// Usage:
25  /// 1. Declare a specific NetEvent with data type to transmit as member of some class.
26  /// 2. Instantiate the NetEvent object in Awake/Init, specifying its exact settings in the constructor:
27  /// * Default data to be sent when invoked without handing data.
28  /// * Target (All, Others, Specifc client)
29  /// * Reliability of the transmission
30  /// * Caching in the room for players yet to join
31  /// 3. Subscribe to the event by AddListener() where you need it.
32  /// 4. Invoke where applicable, handing the data of defined type.
33  ///
34  /// Internally, instances of these classes map the single existing PhotonEvent (identified via ID) to
35  /// local UnityEvents, which can be comfortably subscribed to.
36  public class NetEvent<NetDataType> : IOnEventCallback where NetDataType : INetDataPkg, new() {

101  /// Sends the event to the set targets, calling the registered callbacks with the given data.
102  /// You can optionally hand the id of a user to send only to that user.
103  public void Invoke(NetDataType data, string userID = "") {
104  if (!(NetAdapter.IsConnected() && NetAdapter.IsInRoom())) {
105  Debug.LogWarning("NetEvents can only be invoked while connected/OfflineMode and in a room! Aborting.");
106  return;
107  }
108 
109  // Check whether a specific target was given via userID to retrieve event option values:
110  ReceiverGroup receivers = target.Map2ReceiverGroup();
111  int specificTargetActorNr = GetSpecificTargetActorNr(userID);
112 
113  // Setup with the retrieved values & send the Network message:
114  SendOptions sendOptions = new SendOptions { Reliability = reliable };
115  RaiseEventOptions raiseEventOptions = new RaiseEventOptions();
116 
117  if (specificTargetActorNr != -1) { // a valid actor nr was determined
118  raiseEventOptions.TargetActors = new int[] { specificTargetActorNr };
119  raiseEventOptions.Receivers = ReceiverGroup.All;
120  }
121  else {
122  raiseEventOptions.Receivers = receivers;
123  }
124 
125  if (cachedForRoom) {
126  raiseEventOptions.CachingOption = EventCaching.AddToRoomCache;
127  }
128 
129  PhotonNetwork.RaiseEvent(eventCode, data.ToRaw(), raiseEventOptions, sendOptions);
130  }

163  /// Get the actorNr for a specified userID. Returns -1 if the user cannot be found or was not given.
164  private int GetSpecificTargetActorNr(string userID) {
165  if (userID == "") { // not given
166  return -1;
167  }
168  else if (userID == NetAdapter.GetClientID()) { // Targeted at this client
169  return NetAdapter.GetActorNr();
170  }
171 
172  // Else: a real userID was given, search for it:
173  Client c = NetAdapter.GetClient(userID);
174  if (c == null) {
175  Debug.LogWarning("userID " + userID + " for event target was not found in connected clients! " +
176  "Instead sending accordding to set NetEvent settings: " + this.target);
177  return -1;
178  }
179  else {
180  return c.ActorNumber;
181  }
