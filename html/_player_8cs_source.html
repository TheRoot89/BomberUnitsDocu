51  /// Creates a new Player instance based on incoming NET data. The corresponding event can origin from
52  /// this client or another in the room -> Set Type accordingly.
53  public static Player NewPlayerFromNetData(PlayerFullNetData data) {
54  // The type is not read but inferred from the present clientIDs:
55  if (data.clientID == NetAdapter.GetClientID()) { // Local
56  data.type = Type.local;
57  }
58  else if (NetAdapter.IsClientConnected(data.clientID)) { // Remote
59  data.type = Type.remote;
60  }
61  else { // Erroneous
62  Debug.LogWarning("Player::NewPlayerFromPackedNetData: Received unknown client ID. Returning null!");
63  return null;
64  }
65 
66  ColorScheme color = ColorScheme.FromKey(data.colorDictKey);
67  PlayerFigurePrefab prefab = PlayerFigurePrefab.FromKey(data.playerFigurePrefabDictKey);
68  InputScheme input = InputScheme.FromKey(data.inputDictKey);
69  Player newPlayer = new Player(data.name, color, input, data.type, prefab);
70 
71  newPlayer.clientID = data.clientID;
72  newPlayer.UID = data.UID;

153  /// Updates the details of remote or local players with incoming network data
154  /// For local players, the taken attributes of input and color are changed as well.
155  /// Packed Net format: see Management::Player
156  public bool UpdateWith(PlayerFullNetData data) {
157  if ((clientID != data.clientID)
158  || (UID != data.UID)) {
159  Debug.LogWarning("Player::UpdateWith: Mismatch in existing and given player IDs or Type!");
160  return false;
161  }
162 
163  this.name = data.name;
164 
165  // Update color if changed:
166  if (data.colorDictKey != color.dictKey) {
167  ColorScheme newColor = ColorScheme.FromKey(data.colorDictKey);
168  if (IsLocal()) {
169  color.taken = false;
170  newColor.taken = true;
171  }
172  color = newColor;
173  }
174 
175  // Update figure if changed:
176  if (data.playerFigurePrefabDictKey != playerFigurePrefab.dictKey) {
177  playerFigurePrefab = PlayerFigurePrefab.FromKey(data.playerFigurePrefabDictKey);
178  }
179 
180  // Update input if changed:
181  if (data.inputDictKey != input.dictKey) {
182  InputScheme newInput = InputScheme.FromKey(data.inputDictKey);
183  if (IsLocal()) {
184  input.taken = false;
185  newInput.taken = true;
186  }
187  input = newInput;
188  }
189 
190  return true;
191  }
