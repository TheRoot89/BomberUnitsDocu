175  /// Invoking the event is always the last step to be consistent between direct callbacks during
176  /// OfflineMode and delayed transmission when online.
177  /// A change can mean an addition, a removal or a change in content.
178  /// Local Players are not added/removed via their net Data normally -> issue warnings.
179  /// Also a data change will have occurred due to previous local customization.
180  /// Remote players in turn are signaled to have been added or removed via their net data.
181  private void InterpreteNetDataChangeAndSendEvents(string clientId, int uid, NetData netData) {
182  if (Player.Contains(clientId, uid)) { // Known
183  Player p = Player.Get(clientId, uid);
184  if (netData == null) { // Deletion
185  p.FreeResourcesAndRemove(); // will call the applicable override of Local- or RemotePlayer
186  playerChangeEvent.Invoke(p, EventCause.Deletion);
187  }
188  else { // Data Change or Addition of a Local Player
189  if (clientId == NetAdapter.GetClientID() && ((LocalPlayer)p).isNew) { // is local and new -> Addition
190  ((LocalPlayer)p).isNew = false;
191  playerChangeEvent.Invoke(p, EventCause.Addition);
192  }
193  else { // Change
194  p.OnNetDataChange(netData);
195  playerChangeEvent.Invoke(p, EventCause.Change);
196  }
197  }
198  }
199  else { // Addition of a Player
200  if (clientId == NetAdapter.GetClientID()) { // is local
201  Debug.LogWarning("An unknown local player id should not be received. ClientId: " + clientId);
202  // Local players are already known, as their instances are created before the netData arrives.
203  // Their Addition event is invoked above, checking a flag on them.
204  }
205  else { // is remote -> add local representation
206  Player p = RemotePlayer.CreateNew(clientId, uid, netData);
207  PlayerManager.playerChangeEvent.Invoke(p, EventCause.Addition);
208  }
209  }

214  /// When entering another client's room, this client needs to:
215  /// 1. Add all already existing players in the room
216  /// 2. Check own local players for compatibility.
217  /// 3. Adapt (if needed) local players and publish them to the other clients.
218  /// This is best done in simply re-transmitting the player net data for other to pick up the change
219  /// event. Regarding adding already existing remote players, we only locally re-send the change event
220  /// without actually sending out data of another client.
221  private void OnJoinedRoom() {
222  if (NetManager.GetCurState().name != "JoinRoom") {
223  return;
224  }
225  // Grab all other clients' players and publish locally:
226  foreach (Client c in NetAdapter.GetAllClients()) {
227  string clientId = c.UserId; // called 'user' in Photon, but not consistently
228  if (clientId == NetAdapter.GetClientID()) {
229  continue;
230  }
231 
232  NetManager.clientPropertiesUpdateEvent.Invoke(clientId, NetAdapter.GetClientHashtable(clientId));
233  }

236  /// \todo: Add checks here whether own players can be added at all (resources, nr of players.)
237  /// Do necessary modification to local players if there are conflicts.
